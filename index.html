function sumDigPow(a, b) {
    let res = [];

    for (let i = a; i <= b; i++) {
        let sum = 0;
        let digits = i.toString().split('');

        for (let i = 0; i < digits.length; i++) {
            sum += Math.pow(parseInt(digits[i]), i + 1);
        }

        if (sum === i) {
            res.push(i);
        }
    }

    return res;
}
